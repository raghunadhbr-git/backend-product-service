COMMON VARIABLES (SET MANUALLY AFTER STEPS)
====================================================================

REM 🔑 SELLER JWT (after login)
SET SELLER_TOKEN=PASTE_SELLER_ACCESS_TOKEN_HERE

REM 🔑 CUSTOMER JWT (after login)
SET CUSTOMER_TOKEN=PASTE_CUSTOMER_ACCESS_TOKEN_HERE

REM 🆔 PRODUCT ID (after product creation)
SET PRODUCT_ID=PASTE_PRODUCT_ID_HERE

REM 🆔 VARIANT ID (after variant creation / fetch)
SET VARIANT_ID=PASTE_VARIANT_ID_HERE

====================================================================
AUTH SERVICE
====================================================================

--------------------------------------------------
1️⃣ AUTH SERVICE – HEALTH CHECK
--------------------------------------------------
curl %AUTH_BASE_URL%/api/v1/auth/

--------------------------------------------------
2️⃣ SELLER REGISTRATION
--------------------------------------------------
curl -X POST %AUTH_BASE_URL%/api/v1/auth/angularUser/register ^
-H "Content-Type: application/json" ^
-d "{\"email\":\"seller@test.com\",\"password\":\"seller123\",\"role\":\"seller\"}"

--------------------------------------------------
3️⃣ SELLER LOGIN (COPY access_token)
--------------------------------------------------
curl -X POST %AUTH_BASE_URL%/api/v1/auth/angularUser/login ^
-H "Content-Type: application/json" ^
-d "{\"email\":\"seller@test.com\",\"password\":\"seller123\"}"

REM ⬆ COPY access_token → SET SELLER_TOKEN

--------------------------------------------------
4️⃣ SELLER PROFILE (JWT VERIFY)
--------------------------------------------------
curl -X GET %AUTH_BASE_URL%/api/v1/auth/profile ^
-H "Authorization: Bearer %SELLER_TOKEN%"

====================================================================
PRODUCT SERVICE – CORE
====================================================================

--------------------------------------------------
5️⃣ PRODUCT SERVICE – HEALTH CHECK
--------------------------------------------------
curl %PRODUCT_BASE_URL%/api/angularProduct/health

--------------------------------------------------
6️⃣ SELLER ADDS PRODUCT (JWT REQUIRED)
--------------------------------------------------
curl -X POST %PRODUCT_BASE_URL%/api/angularProduct/add ^
-H "Authorization: Bearer %SELLER_TOKEN%" ^
-H "Content-Type: application/json" ^
-d "{\"name\":\"Laptop\",\"price\":55000,\"description\":\"Gaming laptop\",\"category\":\"Electronics\"}"

REM ⬆ COPY product ID → SET PRODUCT_ID

--------------------------------------------------
7️⃣ SELLER VIEWS ALL PRODUCTS
--------------------------------------------------
curl -X GET %PRODUCT_BASE_URL%/api/angularProduct/get

--------------------------------------------------
8️⃣ SELLER VIEWS SINGLE PRODUCT
--------------------------------------------------
curl -X GET %PRODUCT_BASE_URL%/api/angularProduct/get/%PRODUCT_ID%

====================================================================
PRODUCT VARIANT + STOCK (FEATURE BRANCH)
====================================================================

--------------------------------------------------
9️⃣ CHECK PRODUCT VARIANTS (EXPECT EMPTY INITIALLY)
--------------------------------------------------
curl -X GET %PRODUCT_BASE_URL%/api/angularProduct/get/%PRODUCT_ID%

REM ⬆ From response, note variant IDs when present

--------------------------------------------------
🔟 DECREASE VARIANT STOCK (ORDER PLACED SIMULATION)
--------------------------------------------------
curl -X POST %PRODUCT_BASE_URL%/api/v1/products/decrease-stock ^
-H "Authorization: Bearer %CUSTOMER_TOKEN%" ^
-H "Content-Type: application/json" ^
-d "{
  \"items\": [
    {
      \"product_id\": %PRODUCT_ID%,
      \"variant_id\": %VARIANT_ID%,
      \"quantity\": 2
    }
  ]
}"

EXPECTED:
- HTTP 200
- Stock reduced for variant

--------------------------------------------------
1️⃣1️⃣ VERIFY STOCK DECREASE
--------------------------------------------------
curl -X GET %PRODUCT_BASE_URL%/api/angularProduct/get/%PRODUCT_ID%

EXPECTED:
- Variant stock reduced by quantity

--------------------------------------------------
1️⃣2️⃣ RESTORE VARIANT STOCK (CANCEL / RETURN)
--------------------------------------------------
curl -X POST %PRODUCT_BASE_URL%/api/v1/products/restore-stock ^
-H "Authorization: Bearer %CUSTOMER_TOKEN%" ^
-H "Content-Type: application/json" ^
-d "{
  \"items\": [
    {
      \"product_id\": %PRODUCT_ID%,
      \"variant_id\": %VARIANT_ID%,
      \"quantity\": 2
    }
  ]
}"

EXPECTED:
- HTTP 200
- Stock restored

--------------------------------------------------
1️⃣3️⃣ VERIFY STOCK RESTORE
--------------------------------------------------
curl -X GET %PRODUCT_BASE_URL%/api/angularProduct/get/%PRODUCT_ID%

EXPECTED:
- Variant stock returned to original value
